# ===============frp==============
# 网关穿透, 先搞到frp安装包 解压什么的, 然后 cd frp
# frp server服务端
cp frps /usr/bin/frps

cp frps.ini /etc/frp/frps.ini
cp systemd/frps.service /usr/lib/systemd/system/frps.service
vi /etc/frp/frps.ini
systemctl daemon-reload
systemctl start frps
systemctl enable frps

# frp client客户端

systemctl daemon-reload
systemctl start frpc
systemctl enable frpc

frpc.ini和frps.ini配置文件示例

# frps.ini 配置文件, 服务端口+访问令牌
[common]
bind_port = 7000
token = wx3c7cbe553d341208

# frpc.ini 配置文件
# 目标服务器信息 (必输)
[common]
server_addr = 47.114.82.96
server_port = 7000
token = wx3c7cbe553d341208

[ssh]
# 普通tcp形式
type = tcp
local_ip = 1.1.1.10
local_port = 22
remote_port = 6000

# 唯一name, 作为server服务提供访问
[4570k_mysql]
# 加密形式
type = stcp
# 只有 sk 一致的用户才能访问到此服务
sk = 1
local_ip = 127.0.0.1
local_port = 3306

# 请求访问指定server服务
[4570k_mysql_visitor]
type = stcp
role = visitor
server_name = 4570k_mysql
sk = 1
# 绑定本地端口用于访问 SSH 服务
bind_addr = 127.0.0.1
bind_port = 3306

docker pull nacos/nacos-server
# 先只创建这两个, 后面的会从容器中拷贝出来

# 创建数据库, 并且刷入SQL  schema.sql
# CREATE DATABASE `nacos_config` /*!40100 COLLATE 'utf8mb4_unicode_ci' */;
# 随便启动一下, 将配置文件拷贝出来,
docker run -d --name nacos nacos/nacos-server
docker container cp nacos:/home/nacos/conf/  /mydata/nacos/		# application.properties 	spring启动配置,数据库配置的地方
docker container cp nacos:/home/nacos/init.d/  /mydata/nacos/	# custom.properties 		用户自定义配置文件
docker container cp nacos:/home/nacos/bin/  /mydata/nacos/		# 启动文件在这里面,堆大小配置的地方
# 然后停用,删掉 docker stop ; docker rm ; 别忘了改一下这项配置 management.endpoints.web.exposure.include=*
vi /mydata/nacos/init.d/custom.properties
# 注意ip:MYSQL_SERVICE_HOST, 注意数据库名称MYSQL_SERVICE_DB_NAME
docker run -d -p 8848:8848  \
-e MODE=standalone \
-e PREFER_HOST_MODE=hostname \
-e SPRING_DATASOURCE_PLATFORM=mysql \
-e MYSQL_SERVICE_HOST=1.1.1.101 \
-e MYSQL_SERVICE_PORT=3306 \
-e MYSQL_SERVICE_DB_NAME=nacos_config \
-e MYSQL_SERVICE_USER=root \
-e MYSQL_SERVICE_PASSWORD=root \
-e MYSQL_DATABASE_NUM=1 \
-e JVM_XMS=100m \
-e JVM_XMX=100m \
-e JVM_XMN=50m \
-v /mydata/nacos/init.d/custom.properties:/home/nacos/init.d/custom.properties \
-v /mydata/nacos/logs:/home/nacos/logs \
--restart always --name nacos nacos/nacos-server

Elastic Search和kibana

# 下载 Elasticsearch本体和 可视化界面kibana
docker pull elasticsearch:7.6.2
docker pull kibana:7.6.2

# 简单配置, 开启远程端口访问, 赋予权限
mkdir -p /mydata/elasticsearch/data
mkdir -p /mydata/elasticsearch/config

# 删掉
echo "http.host: 0.0.0.0" >> /mydata/elasticsearch/config/elasticsearch.yml
# 注意:: 为 data 赋予权限, 防止es闪退
chmod -R 777 /mydata/elasticsearch/

# 简单启动 Elasticsearch, 设置内存大小(给够512m), 映射配置, 持久化, 插件
docker run --name elasticsearch -p 9200:9200 -p 9300:9300 \
-e "discovery.type=single-node" \
-e ES_JAVA_OPTS="-Xms64m -Xmx512m" \
-v /mydata/elasticsearch/config/elasticsearch.yml:/usr/share/elasticsearch/config/elasticsearch.yml \
-v /mydata/elasticsearch/data:/usr/share/elasticsearch/data \
-v /mydata/elasticsearch/plugins:/usr/share/elasticsearch/plugins \
-d elasticsearch:7.6.2

# 启动 kibana, 注意这里的 ELASTICSEARCH_HOSTS=http://1.1.1.110:9200 请一定要自行指定ES主机ip
docker run --name kibana -e ELASTICSEARCH_HOSTS=http://1.1.1.110:9200 -p 5601:5601 -d kibana:7.6.2

# 设置容器开机启动
docker update elasticsearch --restart=always
docker update kibana  --restart=always

# 测试: 访问kibana http://nampy.top:5601/app/kibana
# 测试ES:
# ===== 查询 ===== GET =====
# GET /	版本信息
