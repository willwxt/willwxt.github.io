      # ReentrantLock

      # 同步器的实现

      可重入锁中写了3个AQS同步器的实现类: Sync , NonFairSync , FairSync ; 在创建可重入锁的时, 需要选择实例化一个公平与否的同步器

      1. Sync : 继承了AQS的同步器, 已经实现了非公平 tryAquire 模板方法, 解锁, 提供lock模板方法
      2. NonFairSync : 继续继承Sync, tryAquire 直接用父类的抢锁, 自己实现了 lock()
      3. FairSync : 继续继承Sync, 自己实现抢锁(公平), 实现lock()
      4. state: 同步状态; 控制/计数

      • tryAquire 是 AQS 中用于线程争抢锁时回调的模板方法

        • 非公平: 没有人持有锁时尝试CAS state, 有人持有时看看是不是自己
        • 公平: 在非公平实现之前, 判断CLH队列里有没有其他线程

      • tryRelease 释放锁时需要实现的模板方法

        • 当前线程是否为持有线程校验
        • 将 state 减少, 根据state是否归零决定: ①返回解锁成功与否 ②将持有锁的线程设置为null

      • lock流程:

        • 非公平: 先尝试一次 cas state , 失败就进入 acquire
        • 公平: 不会尝试cas, 直接进入 acquire

      # 提供的方法

      可重入锁中提供的方法很多, 实现代码很少, 大都是 AQS 提供的模板方法的组合

      # 加锁类 / 抢锁类

      1. lock 阻塞抢锁, 实际上就调用了 acquire(1)
        • acquire: 主要调用 tryAcquire , 然后是AQS后续逻辑: addWaiter , acquireQueued
      2. lockInterruptibly 可以被interrupt中断的阻塞抢锁
      3. tryLock 就是 tryAcquire
      4. tryLock( timeout ) 保持在一段时间内进行 tryAcquire
      5. lock 和 tryLock 的区别, 等同于 acquire 和 tryAcquire 的区别
        • lock / acquire 是阻塞等待的, 同步
        • tryLock / tryAcquire(1) 是只尝试一次, 如果不行直接返回失败, (异步)
      6. unlock , 解锁, 调用 tryRelease
      7. newCondition

      //1. acquire流程: 先try, 然后addWaiter(创建节点), acquireQueued(节点入队)
public final void acquire(int arg) {
    if (!tryAcquire(arg) && //tryAcquire, 实现类实现
        acquireQueued(	//队列中的线程获取锁
            addWaiter(Node.EXCLUSIVE), //加入队列尾部
            arg
        ))
        selfInterrupt();
}

//2. addWaiter流程, 创建节点并加入队列(尾节点)
private Node addWaiter(Node mode) {
    Node node = new Node(Thread.currentThread(), mode);//创建对应模式的节点
    // 尝试一次快速加入队列, 没有加入成功就调用普通自旋enq(加入队列)
    Node pred = tail;
    if (pred != null) {
        node.prev = pred;
        if (compareAndSetTail(pred, node)) {//尝试将tail设置为本节点(用tailOffset)
            pred.next = node;//这里不会有线程问题, 因为每个线程中tail和pred是不同的
            return node;
        }
    }
    enq(node);//就是自旋加入队列,情况有二: 1.tail为null; 2.别的线程抢占
    return node;
}
//3. enq: (初始化)并加入尾节点
private Node enq(final Node node) {
    for (;;) {
        Node t = tail;
        if (t == null) { // Must initialize
            if (compareAndSetHead(new Node()))//空节点作为头节点
                tail = head;
        } else {
            node.prev = t;
            if (compareAndSetTail(t, node)) {
                t.next = node;
                return t;
            }
        }
    }
}

      # 为什么队列的头结点是个空节点

      CLH队列需要一个虚拟头结点(a dummy header node) , 头结点被认为是已完成同步的节点 , 其中的属性尽量释放以便垃圾回收 , 所以置为null

      # Node的结构

      1. waitStatus 节点状态, 枚举值如下
        • 默认值 0
        • CANCELLED 1, 表示线程获取锁的请求已经取消了
        • CONDITION -2, 表示节点在等待队列中, 节点线程等待唤醒
        • PROPAGATE -3, 当前线程处在SHARED情况下，该字段才会使用
        • SIGNAL -1, 表示线程已经准备好了, 就等资源释放了
      2. prev 和 next 前驱指针, 后继指针; pred(predecessor) 返回前驱节点
      3. thread 处于该节点的线程
      4. nextWaiter 指向下一个处于condition状态的节点
      //acquireQueued 在queue中acquire锁... 真励志
final boolean acquireQueued(final Node node, int arg) {
    boolean failed = true;//标记是否获取资源失败	//获取资源成功标记
    try {
        boolean interrupted = false;//标记等待过程中是否中断过
        for (;;) {//自旋, 要么获得锁, 要么中断
            final Node p = node.predecessor();//前驱节点p	
            if (p == head				//前驱节点p就是头节点? 
                 && tryAcquire(arg)) {	//尝试获取锁, 以下为获取成功的逻辑
                setHead(node);	//把自己置为头节点, 并且将内部数据清空(变为虚拟节点了)
                p.next = null; 	// help GC
                failed = false;	//获取资源成功标记
                return interrupted;
            }
            if (shouldParkAfterFailedAcquire(p, node) //抢锁失败, 判断是否需要park
                && parkAndCheckInterrupt())	//park并检查是否要中断
                interrupted = true;
        }
    } finally {
        if (failed)
            cancelAcquire(node);//如果被中断了, 取消该节点
    }
}
//shouldParkAfterFailedAcquire 检查(并更新)ws状态, 返回是否需要park
private static boolean shouldParkAfterFailedAcquire(Node pred, Node node) {
    int ws = pred.waitStatus;//前驱节点的ws状态
    if (ws == Node.SIGNAL)	//前驱节点已唤醒(-1), 本节点可以park
        return true;
    if (ws > 0) {//前驱节点是已取消(1), 循环向前删除已取消的节点
        do {
            node.prev = pred = pred.prev;
        } while (pred.waitStatus > 0);
        pred.next = node;
    } else {
        //设置前驱节点ws为SIGNAL
        compareAndSetWaitStatus(pred, ws, Node.SIGNAL);
    }
    return false;//本节点不park
}
//parkAndCheckInterrupt park并检查是否中断
private final boolean parkAndCheckInterrupt() {
    LockSupport.park(this);
    return Thread.interrupted();
}
/*cancelAcquire 被中断的节点, 取消该节点在队列中的竞争
- 比较复杂, 概述: 
1. 首先从本节点开始向前一直取到非CANCELLED状态节点作为 pred(前驱节点) (常规操作..
2. 修改本节点状态为CANCELLED
3. 进入判断逻辑, 多条件分析
3.1 本节点是尾节点, 并且将pred设置为尾节点成功, 本方法结束(本节点已删除成功)
3.2 本节点是尾节点, 但是将pred设置尾节点失败了 (说明别的线程修改过,你不再是尾节点)
  或者本节点原来就不是尾节点, 进入下一个判断4.x
4.1 pred不是头节点, 将pred的ws状态设置为SIGNAL唤醒, pred持有线程不为空
  如果next不为空, 让pred的后继指针指向本节点的next
4.2 pred是头结点, 或者持有线程为空/设置ws状态失败(很可能pred被别的线程中断了)
  使用unpark唤醒后继节点
5 为什么没修改prev指针? 可能pred也被移除队列, 一般在shouldParkAfterFailedAcquire中修改prev指针, 该方法在抢锁失败之后执行, 
*/
private void cancelAcquire(Node node) {
    if (node == null) return; // 常规判空
    node.thread = null;	// 直接干掉持有的线程属性, 不持有线程
    Node pred = node.prev; // 通过前驱节点，跳过取消状态的node
    while (pred.waitStatus > 0) node.prev = pred = pred.prev;

    Node predNext = pred.next;	// pred的后继指针next
    node.waitStatus = Node.CANCELLED;	// 修改本节点状态

    // 如果本节点是尾节点 , 将pred设置为尾节点
    if (node == tail && compareAndSetTail(node, pred)) {
        compareAndSetNext(pred, predNext, null); // 将pred的后继指针置为null
    } else {
        int ws;
        if (pred != head && // 前驱节点不是头结点 并且
            ((ws = pred.waitStatus) == Node.SIGNAL || // 前驱节点是唤醒状态或
             // 前驱节点不是取消状态 并且 成功设置为唤醒状态
             (ws <= 0 && compareAndSetWaitStatus(pred, ws, Node.SIGNAL))) &&
            pred.thread != null) { // 并且 前驱节点的持有线程不为空
            Node next = node.next; // 当前节点的后继节点
            if (next != null && next.waitStatus <= 0) // 后继节点不为空+非取消状态
                compareAndSetNext(pred, predNext, next); // 将next设置为前驱节点的next
        } else {
            unparkSuccessor(node); // unpark唤醒后继节点
        }
        node.next = node; // help GC
    }
}

      # 解锁过程 release

      ReentrantLock解锁过程并不区分公平锁和非公平锁

      
public final boolean release(int arg) {
    if (tryRelease(arg)) { // 如果释放成功
        Node h = head;
        if (h != null && h.waitStatus != 0) // 如果头结点不为空, 并且ws不是初始状态
            //h等于null说明还未初始化; 
            //ws==0  说明后继节点对应的线程仍在运行中,不需要唤醒 (为什么? TODO)
            //ws < 0 后继节点可能被阻塞了,需要唤醒 (那确实)
            //ws > 0 呢? 后继节点中断了会怎么样
            unparkSuccessor(h); // unpark唤醒头结点的后继节点
        return true;
    }
    return false;
}

protected final boolean tryRelease(int releases) {
    int c = getState() - releases; // 减少重入锁的次数
    if (Thread.currentThread() != getExclusiveOwnerThread()) // 判断是持有锁的线程吗
        throw new IllegalMonitorStateException();
    boolean free = false; // 是否完全释放标志
    if (c == 0) { // 如果重入锁的次数减少到0, 说明完全释放
        free = true;
        setExclusiveOwnerThread(null);
    }
    setState(c);
    return free;
}

private void unparkSuccessor(Node node) {
    int ws = node.waitStatus;
    if (ws < 0) compareAndSetWaitStatus(node, ws, 0); // 将本节点的状态修改为0
    Node s = node.next;
    if (s == null || s.waitStatus > 0) { // 后继节点为空或被中断了, 找到
        s = null;
        // 从尾部向前找状态不为被中断的节点, 然后取队首第一个, 记为步骤2 (为什么从后往前找?)
        for (Node t = tail; t != null && t != node; t = t.prev)
            if (t.waitStatus <= 0) s = t;
    }
    // 步骤2 找到了这个节点, 就把这个节点unpark唤醒
    if (s != null) LockSupport.unpark(s.thread);
}

      # 为什么要从后往前找node

      1. 因为addWaiter的入队方法不是原子操作(先设置prev, 再CAS, 再设置next), 在CAS设置tail之后, 如果调用unparkSuccessor, 无法从前往后找
      2. cancelAcquire中断节点的时候, 只断开next指针, 没法从前往后找

      # 中断恢复后的执行流程

      //parkAndCheckInterrupt park并检查是否中断
private final boolean parkAndCheckInterrupt() {
    LockSupport.park(this);
    return Thread.interrupted();
}
//acquireQueued
final boolean acquireQueued(final Node node, int arg) {
    boolean failed = true;
    try {
        boolean interrupted = false;//标记等待过程中是否中断过
        for (;;) {//自旋
            final Node p = node.predecessor();
            if (p == head
                 && tryAcquire(arg)) {	//尝试获取锁
                setHead(node);
                p.next = null;
                failed = false;
                return interrupted; // 抢锁成功后, 返回是否被中断了
            }
            if (shouldParkAfterFailedAcquire(p, node) //抢锁失败, 判断是否需要park
                && parkAndCheckInterrupt())	//park并检查是否要中断
                interrupted = true;	// 设置为需要中断, 下次抢锁成功就返回被中断啦
        }
    } finally {
        if (failed)
            cancelAcquire(node);//如果被中断了, 取消该节点
    }
}

public final void acquire(int arg) {
    if (!tryAcquire(arg) &&
        acquireQueued(addWaiter(Node.EXCLUSIVE), arg)) 
        selfInterrupt(); // 被中断就会执行这条语句
}
static void selfInterrupt() {
    Thread.currentThread().interrupt(); // 中断当前线程
}

      final V putVal(K key, V value, boolean onlyIfAbsent) {
    //常规判空
    if (key == null || value == null) throw new NullPointerException();
    //计算hash值, spread: (h ^ (h >>> 16)) & HASH_BITS;
    int hash = spread(key.hashCode());
    int binCount = 0;
    for (Node<K,V>[] tab = table;;) { // 死循环
        Node<K,V> f; int n, i, fh;
        if (tab == null || (n = tab.length) == 0) { // 常规判空
            tab = initTable(); // 初始化
        // 记hash值对应数组下标位置的对象为f,  如果f为null
        }else if ((f = tabAt(tab, i = (n - 1) & hash)) == null) {
            // 创建节点, 尝试cas赋值到空位, cas成功就退出死循环
            if (casTabAt(tab, i, null, new Node<K,V>(hash, key, value, null)))
                break;                   // 当目标槽位为空, 是不需要加锁的!
        // 如果f为MOVED, 说明表格正在扩容
        }else if ((fh = f.hash) == MOVED) {
            tab = helpTransfer(tab, f); // 帮助rehash
        // else, f不为空, 非扩容状态, 
        }else {
            V oldVal = null;
            synchronized (f) 
                //以f作为monitor加锁, 处理f链表/红黑树(就是查找+修改+返回原值)
            }
            if (binCount != 0) { // 0说明没有进行修改(已经有人修改了)
                if (binCount >= TREEIFY_THRESHOLD) { // 链表长度大于树化阈值
                    treeifyBin(tab, i);
                }
                if (oldVal != null) {
                    return oldVal;
                }
                break;
            }
        }
    }
    addCount(1L, binCount); // 增加计数,是否初始化rehash或者帮助rehash
    return null;
}

      private final Node<K,V>[] initTable() {
    Node<K,V>[] tab; int sc;
    while ((tab = table) == null || tab.length == 0) {
        if ((sc = sizeCtl) < 0) { // 说明有别的线程在初始化, sizeCtl默认0
            Thread.yield(); // 自旋
        }else if (U.compareAndSwapInt(this, SIZECTL, sc, -1)) { // 设置-1, CAS取锁
            try {
                if ((tab = table) == null || tab.length == 0) { // 数组为空
                    int n = (sc > 0) ? sc : DEFAULT_CAPACITY; // 常规判空
                    Node<K,V>[] nt = (Node<K,V>[])new Node<?,?>[n];
                    table = tab = nt;
                    sc = n - (n >>> 2); // 设置sc, 容量为16的话就设置为12
                }
            } finally {sizeCtl = sc;}
            break;
        }
    }
    return tab;
}

      # helpTransfer在rehash过程中帮助转移

      final Node<K,V>[] helpTransfer(Node<K,V>[] tab, Node<K,V> f) {
    Node<K,V>[] nextTab; int sc;
    if (tab != null && (f instanceof ForwardingNode) &&
        (nextTab = ((ForwardingNode<K,V>)f).nextTable) != null) {
        int rs = resizeStamp(tab.length);
        while (nextTab == nextTable && table == tab &&
               (sc = sizeCtl) < 0) {
            if ((sc >>> RESIZE_STAMP_SHIFT) != rs || sc == rs + 1 ||
                sc == rs + MAX_RESIZERS || transferIndex <= 0)
                break;
            if (U.compareAndSwapInt(this, SIZECTL, sc, sc + 1)) {
                transfer(tab, nextTab);
                break;
            }
        }
        return nextTab;
    }
    return table;
}

      # 特殊节点: ForwardingNode

      static final class ForwardingNode<K,V> extends Node<K,V> { // 转移过程中插入bins头部
    final Node<K,V>[] nextTable;
    ForwardingNode(Node<K,V>[] tab) {
        super(MOVED, null, null, null); // 创建一个 -1 的节点放在头部
        this.nextTable = tab;
    }
    Node<K,V> find(int h, Object k) {
        // loop to avoid arbitrarily deep recursion on forwarding nodes
        outer: for (Node<K,V>[] tab = nextTable;;) {
            Node<K,V> e; int n;
            if (k == null || tab == null || (n = tab.length) == 0 ||
                (e = tabAt(tab, (n - 1) & h)) == null)
                return null;
            for (;;) {
                int eh; K ek;
                if ((eh = e.hash) == h &&
                    ((ek = e.key) == k || (ek != null && k.equals(ek))))
                    return e;
                if (eh < 0) {
                    if (e instanceof ForwardingNode) {
                        tab = ((ForwardingNode<K,V>)e).nextTable;
                        continue outer;
                    }
                    else
                        return e.find(h, k);
                }
                if ((e = e.next) == null)
                    return null;
            }
        }
    }
}
